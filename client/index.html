架伺服器

我們聊天室的機器會用 Node 架
這台機器要會能夠

  1. 送網頁
  2. 讓人家聊天

所以我們要先架個網頁伺服器
然後在加上一些功能讓人家能夠聊天

Express

想要架個網頁伺服器可以自己用 http 模組從頭開始寫
但是要寫你自己寫，我要去用人家寫好的 :D

Express是一套網頁伺服器，像 Apache 那種東西
要做 routing、要渲染模版、要插 middleware 什麼的想幹嘛都可以

安裝 Express

                        
npm install express

裝好之後就可以直接用了 

                        
// app.js
var express = require('express');
var http    = require('http');
var path    = require('path');

var app = express();

app.configure(function () {
    app.use(express.static(path.join(__dirname, './public')));
});

var server = http.createServer(app)

server.listen(4000);

加入 socket.io

這時候我們要幫我們的 express 掛上 socket.io

安裝 socket.io

                        
npm install socket.io

然後多加個兩行 

                        
// app.js
var express = require('express');
var http    = require('http');
var path    = require('path');
var io      = require('socket.io'); // <= 這行

var app = express();

app.configure(function () {
    app.use(express.static(path.join(__dirname, './public')));
});

var server = http.createServer(app);

io = io.listen(server); // <= 還有這行

server.listen(4000);

重開看看，沒有噴 error 的話應該就是沒有爆炸了 

                        
node app.js

// app.js

//
//  Express
//

var express = require('express');
var http    = require('http');
var path    = require('path');
var io      = require('socket.io');

var app = express();

app.configure(function () {
    app.use(express.static(path.join(__dirname, './public')));
});

var server = http.createServer(app);

io = io.listen(server);

server.listen(4000);

//
//  Socket.io
//

io.sockets.on('connection', function (socket) {
    console.log('有人連過來了!!');

    socket.emit('hello', '我是伺服器！');

})

後端打電話過來前端也要有人接

// app.js

//
//  Express
//

var express = require('express');
var http    = require('http');
var path    = require('path');
var io      = require('socket.io');

var app = express();

app.configure(function () {
    app.use(express.static(path.join(__dirname, './public')));
});

var server = http.createServer(app);

io = io.listen(server);

server.listen(4000);

//
//  Socket.io
//

io.sockets.on('connection', function (socket) {
    console.log('有人連過來了!!');

    socket.broadcast.emit('有其他人連進來了!');

})
